import java.awt.*;
import java.awt.event.*;
import javax.swing.*;

public class ButtonExample {
    private static boolean isAnimating = true; // Controls animation state

    public static void main(String[] args) {
        JFrame f = new JFrame("Button Example with Animation Control");
        final JTextField tf = new JTextField();
        tf.setBounds(50, 50, 200, 20);

        JButton messageButton = new JButton("Click Here");
        messageButton.setBounds(50, 100, 120, 30);
        messageButton.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                tf.setText("Welcome to Javatpoint.");
            }
        });

        JButton controlButton = new JButton("Stop Animation");
        controlButton.setBounds(200, 100, 150, 30);

        // Panel for animation
        JPanel animationPanel = new JPanel() {
            private int x = 0;

            @Override
            protected void paintComponent(Graphics g) {
                super.paintComponent(g);
                g.setColor(Color.RED);
                g.fillOval(x, 50, 30, 30); // Draw a moving circle
            }

            public void animate() {
                x += 10;
                if (x > getWidth()) x = 0;
                repaint();
            }
        };

        animationPanel.setBounds(50, 150, 300, 200);
        animationPanel.setBackground(Color.LIGHT_GRAY);

        // Thread for animation
        Thread animationThread = new Thread(() -> {
            while (true) {
                if (isAnimating) {
                    animationPanel.animate();
                }
                try {
                    Thread.sleep(100); // Adjust speed of animation
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
        });

        animationThread.start();

        controlButton.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                isAnimating = !isAnimating;
                controlButton.setText(isAnimating ? "Stop Animation" : "Continue Animation");
            }
        });

        f.add(messageButton);
        f.add(controlButton);
        f.add(animationPanel);
        f.add(tf);
        f.setSize(400, 400);
        f.setLayout(null);
        f.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        f.setVisible(true);
    }
}
